\section{Tóm tắt}
%% ============================================================

Báo cáo này trình bày quá trình nâng cấp mô phỏng MU-MIMO hiệu năng NR Cell
từ 3GPP Release~15/16 (gNB 32 cổng) lên Release~19 (gNB 128 cổng, UE 4 cổng).
Công việc bao gồm mở rộng MATLAB~5G~Toolbox---cụ thể là hai tệp thư viện nội
bộ---để hỗ trợ \emph{Codebook Type~I Single-Panel Tinh chỉnh} được định nghĩa
trong TS~38.214 \S5.2.2.2.1a (Rel-19).

\medskip
\noindent\textbf{Kết quả đạt được:}
\begin{itemize}
  \item Xây dựng lưới tài nguyên CSI-RS 128 cổng (4 tài nguyên $\times$ 32 cổng,
        Row~18 CDM8, 2 slot).
  \item Ước lượng kênh đầy đủ 128 cổng (NMSE~$\approx -31{,}5$~dB tại SNR~20~dB,
        CDL-C; $-27{,}8$~dB cho CDL-B).
  \item Báo cáo PMI Rel-19 modeA: lựa chọn RI từ 1 đến 4, SINR và CQI theo từng tầng.
  \item Chứng minh \textbf{RI~=~4} với kênh CDL-B (giá trị kỳ dị gần đều),
        xác nhận codebook hỗ trợ ghép kênh không gian đầy hạng
        lên đến $\min(N_\text{Tx},N_\text{Rx}) = 4$ tầng.
  \item So sánh dung lượng: cận trên SVD~37,3~bits/s/Hz so với
        Mode-A codebook~14,5~bits/s/Hz tại RI~=~4.
\end{itemize}

\section{Bối cảnh và Động lực}
%% ============================================================

\subsection{Giới hạn của Rel-15/16}

MATLAB~5G~Toolbox hiện tại (\texttt{nrPMIReport}) cài đặt codebook Type~I
Single-Panel của TS~38.214 \S5.2.2.2.1 cho tối đa 32 cổng CSI-RS.
Đối với gNB 128 cổng, Rel-15/16 không hỗ trợ codebook trực tiếp.

\subsection{Mở rộng Codebook Rel-19}

3GPP Release~19 (TS~38.214 v19.2.0) giới thiệu \S5.2.2.2.1a,
\emph{Refined Codebook Type~I Single-Panel}, hỗ trợ 48, 64 và 128 cổng anten
thông qua tổng hợp tài nguyên CSI-RS. Với 128~cổng:
\begin{align}
  N_1 \times N_2 \times 2\,\text{phân cực} &= 128, \quad
  (N_1,N_2) \in \{(16,4),\;(8,8)\}, \quad O_1 = O_2 = 4.
\end{align}
Bộ tiền mã hoá dựa trên lưới DFT 2 chiều được oversampling (modeA/modeB).

\subsubsection{Hàm mới: \texttt{getPMIType1SinglePanelR19Codebook()} (dòng~2208)}

Hàm này cài đặt Bảng~5.2.2.2.1a-4 (modeA) của TS~38.214 cho hạng 1--4.
Tensor codebook có kích thước:
\begin{equation}
  \mathbf{C} \in \mathbb{C}^{P \times r \times N_{i_2} \times N_{i_{1,1}} \times N_{i_{1,2}} \times N_{i_{1,3}}}
\end{equation}
với $P=128$, $r$ là rank, và kích thước chỉ số được cho trong
Bảng~5.2.2.2.1a-2.

\paragraph{Vector chùm tia $v_{l,m}$}
\begin{equation}
  v_{l,m} = \mathbf{u}_h(l) \otimes \mathbf{u}_v(m), \quad
  [\mathbf{u}_h(l)]_k = e^{j2\pi l k / (O_1 N_1)}, \quad
  [\mathbf{u}_v(m)]_k = e^{j2\pi m k / (O_2 N_2)}
\end{equation}

\paragraph{Bộ tiền mã hoá RI = 1 ($n = i_2 \in \{0,1,2,3\}$)}
\begin{equation}
  W^{(1)} = \frac{1}{\sqrt{P}}
  \begin{bmatrix} v_{l,m} \\ \varphi(n)\,v_{l,m} \end{bmatrix}
\end{equation}

\paragraph{Bộ tiền mã hoá RI = 2 ($n = i_2 \in \{0,1\}$, offset $(k_1,k_2)$ từ $i_{1,3}$)}
\begin{equation}
  W^{(2)} = \frac{1}{\sqrt{2P}}
  \begin{bmatrix}
    v_{l,m} & v_{l',m'} \\
    \varphi(n)\,v_{l,m} & -\varphi(n)\,v_{l',m'}
  \end{bmatrix}, \quad
  l' = l+k_1 \pmod{N_1 O_1}, \; m' = m+k_2 \pmod{N_2 O_2}
\end{equation}

\paragraph{Bộ tiền mã hoá RI = 4}
\begin{equation}
  W^{(4)} = \frac{1}{\sqrt{4P}}
  \begin{bmatrix}
    v_0 & v_1 & v_2 & v_3 \\
    \varphi(n)\,v_0 & \varphi(n)\,v_1 & \varphi(n)\,v_2 & \varphi(n)\,v_3
  \end{bmatrix}
\end{equation}
với $v_k = v_{l+k k_1,\;m+k k_2}$ cho $k=0,1,2,3$.

\paragraph{Offset chùm tia (Bảng~5.2.2.2.1a-3, $N_2 > 1$)}
\begin{equation}
  i_{1,3} \in \{0,1,2,3\} \;\Rightarrow\;
  (k_1,k_2) \in \{(O_1,0),\;(0,O_2),\;(O_1,O_2),\;(0,2O_2)\}
\end{equation}

\subsection{Xây dựng Ma trận Kênh Wideband}

Mỗi ước lượng kênh theo tài nguyên
$\hat{\mathbf{H}}_r \in \mathbb{C}^{K \times L \times 4 \times 32}$
được trung bình theo tần số và thời gian, rồi ghép lại thành:
\begin{equation}
  \mathbf{H}_\text{wb} \in \mathbb{C}^{4 \times 128}, \quad
  [\mathbf{H}_\text{wb}]_{:,\,pS:pE} =
    \mathbb{E}_{k,l}\left[\hat{\mathbf{H}}_r(k,l,:,:)\right]
\end{equation}
Sau đó broadcast sang $[K \times L \times 4 \times 128]$ cho \texttt{nrPMIReport}.

\subsection{Lựa chọn RI theo Tiêu chí Dung lượng}

RI được chọn bằng cách tối đa hoá tổng tốc độ dữ liệu khả đạt (miền log), \emph{không phải} tổng SINR tuyến tính.
\begin{lstlisting}
for ri_try = 1:min(nRxAntennas, 4)
    [~, info_try] = nr5g.internal.nrPMIReport(..., ri_try, ...);
    sinr_vals = info_try.SINRPerREPMI(:);
    rate_sum  = sum(log2(1 + sinr_vals(isfinite(sinr_vals)...
                                       & sinr_vals >= 0)));
    if rate_sum > best_rate
        best_rate = rate_sum;
        best_ri   = ri_try;
    end
end
\end{lstlisting}

\section{Kết luận}
%% ============================================================

Hệ thống mô phỏng đã được nâng cấp thành công từ Rel-15/16 lên
Release~19 với các kết quả đã được kiểm chứng sau:

\begin{enumerate}
  \item \textbf{CSI-RS 128 Port}: Tổng hợp tài nguyên (4$\times$32~port,
        Row~18 CDM8) tạo ra ước lượng kênh 128 port chính xác tại
        NMSE~$< -27$~dB.

  \item \textbf{Codebook Rel-19}: Refined Codebook Type~I Single-Panel
        (TS~38.214 \S5.2.2.2.1a, modeA) đã được cài đặt vào thư viện workspace
        MATLAB với đầy đủ tuân thủ tiêu chuẩn cho RI~=~1--4.

  \item \textbf{RI~=~4 đã chứng minh}: Với kênh CDL-B (giá trị eigenvectors gần đều),
        codebook modeA Rel-19 chính xác chọn RI~=~4 và báo cáo
        SINR theo tầng $\approx 12$--17~dB tại SNR~20~dB.

  \item \textbf{Không làm hỏng Toolbox}: Tất cả thay đổi được cô lập trong
        các bản sao workspace; MATLAB 5G Toolbox đã cài đặt không bị ảnh hưởng.
\end{enumerate}

\noindent\textbf{Các bước tiếp theo:}
\begin{itemize}
  \item Cài đặt modeB (tìm kiếm chùm độc lập theo từng tầng) để cải thiện
        hiệu năng đa tầng trên các kênh ít hạng (ví dụ CDL-C).
